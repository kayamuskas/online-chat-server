export class ContactsService {

**Required methods:**

1. `async sendFriendRequest(callerId: string, input: { targetUsername: string; message?: string }): Promise<FriendRequest>`
   - Look up target: `const target = await this.userRepo.findByUsername(input.targetUsername)`
   - If !target → `throw new NotFoundException(\`User '\${input.targetUsername}' is not registered\`)`
   - If callerId === target.id → `throw new BadRequestException('Cannot send a friend request to yourself')`
   - Check existing: `const existing = await this.repo.findFriendRequest(callerId, target.id)`
   - If existing → `throw new ConflictException('You already have a pending request to this user')`
   - Call `this.repo.createFriendRequest({ requester_id: callerId, target_id: target.id, message: input.message ?? null })`
   - Return the created FriendRequest

2. `async getIncomingRequests(userId: string): Promise<IncomingFriendRequestView[]>`
   - Return `this.repo.listIncomingRequests(userId)`

3. `async getOutgoingRequests(userId: string): Promise<FriendRequest[]>`
   - Return `this.repo.listOutgoingRequests(userId)`

4. `async acceptRequest(requestId: string, callerId: string): Promise<Friendship>`
   - Find request: `const req = await this.repo.findRequestById(requestId)`
   - If !req → `throw new NotFoundException('Friend request not found')`
   - If req.target_id !== callerId → `throw new ForbiddenException('Only the request recipient can accept it')`
   - If req.status !== 'pending' → `throw new BadRequestException('Request is no longer pending')`
   - Create friendship: `const friendship = await this.repo.createFriendship(callerId, req.requester_id)`
   - Update request status: `await this.repo.updateRequestStatus(requestId, 'accepted')`
   - Return friendship

5. `async declineRequest(requestId: string, callerId: string): Promise<void>`
   - Find request; check target_id === callerId; check status === 'pending'
   - `await this.repo.updateRequestStatus(requestId, 'declined')`

6. `async cancelRequest(requestId: string, callerId: string): Promise<void>`
   - Find request; if req.requester_id !== callerId → `throw new ForbiddenException('Only the sender can cancel a request')`
   - If req.status !== 'pending' → `throw new BadRequestException('Request is no longer pending')`
   - `await this.repo.updateRequestStatus(requestId, 'cancelled')`

7. `async getFriends(userId: string): Promise<FriendWithPresence[]>`
   - Return `this.repo.listFriends(userId)` (presenceStatus defaults to 'offline' in DB; Phase 6 can enrich)

8. `async removeFriend(callerId: string, targetUserId: string): Promise<void>`
   - `const removed = await this.repo.deleteFriendship(callerId, targetUserId)`
   - If !removed → throw NotFoundException('Friendship not found')
   - Note: DM history is NOT frozen (D-19), only eligibility is lost

9. `async banUser(callerId: string, targetId: string): Promise<void>`
   - Self-ban guard: if callerId === targetId → `throw new BadRequestException('Cannot ban yourself')`
   - Use a manual transaction via getClient():
   ```typescript
   const client = await this.db.getClient();
   try {
     await client.query('BEGIN');
     await this.repo.deleteFriendship(callerId, targetId, client);
     await this.repo.createBan({ banner_user_id: callerId, banned_user_id: targetId }, client);
     await this.repo.freezeDmConversation(callerId, targetId, client);
     await client.query('COMMIT');
   } catch (err) {
     await client.query('ROLLBACK');
     throw err;
   } finally {
     client.release();
   }
   ```

10. `async getMyBans(userId: string): Promise<UserBan[]>`
    - Return `this.repo.listBans(userId)`

11. `async unbanUser(callerId: string, targetId: string): Promise<void>`
    - `const ban = await this.repo.findBanByBanner(callerId, targetId)`
    - If !ban → `throw new NotFoundException('Ban not found')`
    - `await this.repo.removeBan(callerId, targetId)`

12. `async checkDmEligibility(callerId: string, targetId: string): Promise<DmEligibilityResult>`
    - `const friendship = await this.repo.findFriendship(callerId, targetId)`
    - If !friendship → return `{ eligible: false, reason: 'not_friends' }`
    - `const ban = await this.repo.findBanBetween(callerId, targetId)`
    - If ban → return `{ eligible: false, reason: 'ban_exists' }`
    - Return `{ eligible: true }`

13. `async initiateDm(callerId: string, targetId: string): Promise<{ conversation: DmConversation; eligible: boolean }>`
    - Call checkDmEligibility; if !eligible → throw `new ForbiddenException(\`DM not allowed: \${eligibility.reason}\`)`
    - `const conversation = await this.repo.createDmConversation(callerId, targetId)`
    - Return `{ conversation, eligible: true }`
  </action>
