2. Add a password strength helper function (simple heuristic per RESEARCH.md):
```typescript
function checkPasswordStrength(pw: string): "strong" | "weak" {
  if (pw.length >= 12) return "strong";
  const hasUpper = /[A-Z]/.test(pw);
  const hasLower = /[a-z]/.test(pw);
  const hasDigit = /[0-9]/.test(pw);
  if (pw.length >= 8 && hasUpper && hasLower && hasDigit) return "strong";
  return "weak";
}
```

5. Replace the catch block to parse field-specific errors per D-9.1-04:
```typescript
catch (err) {
  const msg = err instanceof Error ? err.message : "";
  if (msg.includes("email is already registered")) {
    setEmailStatus("taken");
    // Username was not checked by backend (email fails first) — don't show status
    setUsernameStatus(null);
  } else if (msg.includes("username is already taken")) {

  } else {

  }
}
```
