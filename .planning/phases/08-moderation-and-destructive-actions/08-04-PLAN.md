```typescript
/**
 * Delete the user's account with full cascade (D-10, D-15):
 *   1. Verify password (D-10)
 *   2. Delete owned rooms — each triggers room:deleted WS broadcast + full cascade (D-11)
 *   3. Strip admin roles in non-owned rooms (D-12)
 *   4. Remove remaining memberships
 *   5. Delete contacts/friendships/bans
 *   6. Delete DM conversations — NOT messages (D-13)
 *   7. Delete all sessions (D-14)
 *   8. Delete user record — FK ON DELETE SET NULL preserves DM messages
 */
async deleteAccount(userId: string, password: string): Promise<void> {
  // D-10: verify password before proceeding
  const user = await this.users.findById(userId);
  if (!user) throw new UnauthorizedException('User not found');

  const valid = await verifyPassword(password, user.password_hash);
  if (!valid) throw new UnauthorizedException('Incorrect password');

  // D-11: delete owned rooms (each with WS broadcast + full cascade)
  const ownedRooms = await this.roomsService.listOwnedRooms(userId);
  for (const room of ownedRooms) {
    await this.roomsService.deleteRoom(room.id, userId);
  }

  // D-12: strip admin roles in non-owned rooms
  await this.roomsRepo.removeAdminFromAllRooms(userId);

  // D-15: remove remaining memberships
  await this.roomsRepo.removeMemberFromAllRooms(userId);

  // D-15: delete contacts/friendships/bans
  await this.contactsRepo.deleteAllFor(userId);

  // D-13, D-15: delete DM conversations (NOT messages — messages preserved with author_id SET NULL)
  await this.contactsRepo.deleteDmConversationsFor(userId);

  // D-14: delete all sessions (WS sockets get 401 on next auth check)
  await this.sessions.deleteAllByUserId(userId);

  // Final: delete user record (FK ON DELETE SET NULL handles remaining message/attachment refs)
  await this.users.deleteById(userId);
}
```

```typescript
/**
 * DELETE /api/v1/auth/account
 * Permanently deletes the authenticated account. Requires password confirmation (D-10).
 */
export async function deleteAccount(params: { password: string }): Promise<void> {
  const res = await fetch(`${BASE_URL}/auth/account`, {
    method: "DELETE",
    headers: { "Content-Type": "application/json" },
    credentials: "include",
    body: JSON.stringify(params),
  });
  if (!res.ok) {
    const data = await res.json().catch(() => ({}));
    const msg = typeof data?.message === "string" ? data.message : "Deletion failed. Please try again.";
    throw Object.assign(new Error(msg), { statusCode: res.status });
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> auth API | Password must be verified before destructive cascade |
| auth.service -> rooms cascade | Each owned room must broadcast WS before deletion |
| auth.service -> user DB | FK ON DELETE SET NULL must be in place (Plan 01 prerequisite) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-09 | Spoofing | auth.controller.ts DELETE /account | mitigate | Password verification via `verifyPassword()` before any cascade (D-10); CurrentUserGuard ensures authenticated session |
| T-08-10 | Elevation of Privilege | auth.service.ts deleteAccount | mitigate | Only the authenticated user can delete their own account; no admin can delete another user's account. Controller uses `ctx.user.id` from session. |
| T-08-11 | Denial of Service | auth.service.ts deleteAccount cascade | accept | Cascade may take time for users with many owned rooms. Acceptable since it's a one-time destructive action initiated by the user. No rate-limiting needed beyond authentication. |
| T-08-12 | Information Disclosure | auth.controller.ts error response | mitigate | Wrong password returns 401 with generic "Incorrect password" message. Does not reveal whether user exists (already authenticated). |
| T-08-13 | Repudiation | session cleanup | mitigate | `deleteAllByUserId()` removes ALL sessions including the current one (D-14). WS sockets get 401 on next auth check. `clearSessionCookie(res)` removes cookie on the requesting client. |
</threat_model>
