4. ADD these hooks inside the function body, BEFORE the return:
   ```typescript
   const scrollRef = useRef<HTMLDivElement>(null);
   const [isScrolledUp, setIsScrolledUp] = useState(false);

   function handleScroll() {
     const el = scrollRef.current;
     if (!el) return;
     const distFromBottom = el.scrollHeight - el.scrollTop - el.clientHeight;
     const scrolledUp = distFromBottom > 100;
     setIsScrolledUp(scrolledUp);
     // When user scrolls back to bottom, clear the indicator
     if (!scrolledUp) {
       onScrollToBottom?.();
     }
   }

   // Auto-scroll to bottom when new messages arrive, but only if not scrolled up
   useEffect(() => {
     if (!isScrolledUp && scrollRef.current) {
       scrollRef.current.scrollTo({ top: scrollRef.current.scrollHeight });
     }
   }, [messages.length, isScrolledUp]);
   ```

5. MODIFY the outer div (currently `<div className="msg-timeline" role="log" ...>`):
   ```tsx
   <div
     className="msg-timeline"
     role="log"
     aria-live="polite"
     aria-label="Message history"
     ref={scrollRef}
     onScroll={handleScroll}
     style={{ position: "relative", overflowY: "auto", height: "100%" }}
   >
   ```

6. ADD the floating indicator button as the FIRST child inside the outer div,
