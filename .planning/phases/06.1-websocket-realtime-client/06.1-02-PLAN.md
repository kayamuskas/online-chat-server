/**
 * socket.ts — singleton Socket.IO client module (Phase 6.1).
 *
 * A single socket is shared across the whole React application.
 * Connect/disconnect are driven by auth state (SocketProvider in App.tsx).
 *
 * Why singleton (not React state):
 *   - Prevents multiple io() calls across React renders.
 *   - Socket lifecycle is orthogonal to render cycle.
 *   - React Context (SocketProvider) propagates the reference to components.
 *
 * Auth model:
 *   - `withCredentials: true` sends the "session" HttpOnly cookie on handshake.
 *   - Backend ws-auth.ts validates the cookie; unauthenticated sockets are
 *     disconnected immediately by both AppGateway and MessagesGateway.
 *
 * CORS note (D-01 / RESEARCH.md Critical Finding):
 *   Both gateways must have cors: { origin: 'http://localhost:4173', credentials: true }.
 *   Fixed in plan 06.1-01. Without the fix this module will get connect_error.
 */

import { io, Socket } from "socket.io-client";
import { SERVICE_PORTS } from "@chat/shared";

let socket: Socket | null = null;

/**
 * Connect to the Socket.IO server.
 * Idempotent: returns the existing socket if it is already connected.
 * Always uses withCredentials: true so the session cookie is sent on handshake.
 */
export function connectSocket(): Socket {
  if (socket?.connected) return socket;
  socket = io(`http://localhost:${SERVICE_PORTS.apiHttp}`, {
    withCredentials: true,   // required: session cookie "session" auth (ws-auth.ts)
    autoConnect: true,
  });
  return socket;
}

/**
 * Returns the current socket instance or null if not connected yet.
 * Components use this via useSocket() from SocketProvider.
 */
export function getSocket(): Socket | null {
  return socket;
}

/**
 * Disconnect and nullify the singleton socket.
 * Called on sign-out from SocketProvider.
 * Null-safe: no-op if socket is already null.
 */
export function disconnectSocket(): void {
  socket?.disconnect();
  socket = null;
}
```

Do NOT import connectSocket or getSocket in this file — they are exported for consumers.
Do NOT use autoConnect: false — the server disconnects unauthenticated sockets,
so connecting immediately after auth is the correct behavior.

/**
 * SocketProvider — thin React context wrapper for the singleton socket.
 *
 * Connects the socket when `authenticated` becomes true; disconnects on false.
 * Uses useRef (NOT useState) to hold the socket reference so that
 * connect/disconnect do NOT trigger a full React tree re-render.
 *
 * Usage:
 *   <SocketProvider authenticated={!!user}>
 *     {children}
 *   </SocketProvider>
 *
 *   // In any descendant:
 *   const socket = useSocket();
 *   if (!socket) return;  // not authenticated yet
 *   socket.emit("joinRoom", { roomId });
 */

import { createContext, useContext, useEffect, useRef } from "react";
import type { Socket } from "socket.io-client";
import { connectSocket, disconnectSocket } from "../../lib/socket";

const SocketContext = createContext<Socket | null>(null);

interface SocketProviderProps {
  authenticated: boolean;
  children: React.ReactNode;
}

export function SocketProvider({ authenticated, children }: SocketProviderProps) {
  const socketRef = useRef<Socket | null>(null);

  useEffect(() => {
    if (authenticated) {
      socketRef.current = connectSocket();
    } else {
      disconnectSocket();
      socketRef.current = null;
    }
    // No cleanup needed: disconnectSocket() is idempotent;
    // the socket lifetime is intentionally longer than this effect.
  }, [authenticated]);

  return (
    <SocketContext.Provider value={socketRef.current}>
      {children}
    </SocketContext.Provider>
  );
}

/**
 * Returns the live Socket instance, or null when not authenticated.
 * Must be called inside a SocketProvider descendant.
 */
export function useSocket(): Socket | null {
  return useContext(SocketContext);
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| SocketProvider → SocketContext | socket reference propagated via React context; null before auth |
| connectSocket() → socket.io Manager | single io() call; subsequent calls are no-ops if connected |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06.1-04 | Spoofing | connectSocket() called before auth | mitigate | SocketProvider only calls connectSocket() when `authenticated === true`; App.tsx only renders SocketProvider for authenticated users |
| T-06.1-05 | Denial | Multiple io() calls creating multiple connections | mitigate | `if (socket?.connected) return socket;` guard in connectSocket() prevents multiple connections on re-renders |
| T-06.1-06 | Info Disclosure | Socket persisting after sign-out | mitigate | SocketProvider useEffect on `authenticated=false` calls disconnectSocket(); socket.null-ed immediately |
| T-06.1-07 | Tampering | XSS accessing socket via React context | accept | Socket carries session cookie (HttpOnly, not accessible to JS); socket itself is a transport, not a credential; XSS mitigation is upstream of this phase |
</threat_model>
