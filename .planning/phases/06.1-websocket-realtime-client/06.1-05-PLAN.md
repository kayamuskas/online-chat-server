Presence polling pattern (RESEARCH.md Pattern 6):
```typescript
useEffect(() => {
  if (!socket || contacts.length === 0) return;

  const userIds = contacts.map(c => c.userId);

  function poll() {
    socket!.emit("getPresence", { userIds });
  }

  function onPresence(presenceMap: Record<string, { status: string }>) {
    onPresenceUpdate?.(presenceMap);
  }

  socket.on("presence", onPresence);
  poll();                                    // immediate on mount
  const interval = setInterval(poll, 30_000); // D-03: 30s

  return () => {
    socket.off("presence", onPresence);
    clearInterval(interval);
  };
}, [socket, contacts.map(c => c.userId).join(",")]);
// NOTE: join() creates a stable string dep — avoids infinite loop from array identity change
```
