Reconnect pattern (D-06: 500ms debounce, NO user-visible spinner):
```typescript
function scheduleReconnectRefetch() {
  if (reconnectTimerRef.current) clearTimeout(reconnectTimerRef.current);
  reconnectTimerRef.current = setTimeout(async () => {
    try {
      const result = await getRoomHistory(roomId);  // last page, no cursor
      setMessages(prev => {
        const existingIds = new Set(prev.map(m => m.id));
        const newMsgs = result.messages.filter(m => !existingIds.has(m.id));
        if (newMsgs.length === 0) return prev;
        return [...prev, ...newMsgs].sort(
          (a, b) => a.conversationWatermark - b.conversationWatermark
        );
      });
    } catch {
      // silent — user sees stale history (D-06: no error state)
    }
  }, 500);
}
```
